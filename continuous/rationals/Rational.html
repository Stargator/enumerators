Rational class 

class Rational {
 final int numerator;
 final int denominator;

 static _gcd(int n, int m) {
   while (m != 0) {
     int tmp = m;
     m = n % m;
     n = tmp;
   }
   return n;
 }

 Rational._normalized(this.numerator, this.denominator) {
   assert(denominator > 0);
   assert(_gcd(numerator.abs(), denominator.abs()) == 1);
 }

 factory Rational(int numerator, [int denominator = 1]) {
   if (denominator == 0) throw new ArgumentError('division by zero');
   if (numerator == 0) return new Rational._normalized(0, 1);
   if (denominator < 0) {
     numerator = -numerator;
     denominator = -denominator;
   }
   final aNumerator = numerator.abs();
   final aDenominator = denominator.abs();
   final gcd = _gcd(aNumerator, aDenominator);
   return (gcd == 1)
       ? new Rational._normalized(numerator, denominator)
       : new Rational._normalized(numerator ~/ gcd, denominator ~/ gcd);
 }

 String toString() {
   if (numerator == 0) return '0';
   if (denominator == 1) return numerator.toString();
   else return '$numerator/$denominator';
 }

 int get hashCode => numerator + 31 * denominator;

 bool operator ==(other) {
   if (other is Rational) {
     return numerator == other.numerator && denominator == other.denominator;
   } else if (other is int) {
     return numerator == other && denominator == 1;
   } else if (other is double) {
     return toDouble() == other;
   } else {
     throw new ArgumentError("expecting int, double or Rational");
   }
 }

 bool operator <(other) {
   if (other is Rational) {
     return numerator * other.denominator < other.numerator * denominator;
   } else if (other is int) {
     return numerator < other * denominator;
   } else if (other is double) {
     return toDouble() < other;
   } else {
     throw new ArgumentError("expecting int, double or Rational");
   }
 }

 bool operator >(other) {
   if (other is Rational) {
     return numerator * other.denominator > other.numerator * denominator;
   } else if (other is int) {
     return numerator > other * denominator;
   } else if (other is double) {
     return (numerator / denominator) > other;
   } else {
     throw new ArgumentError("expecting int, double or Rational");
   }
 }

 bool operator <=(other) => this < other || this == other;

 bool operator >=(other) => this > other || this == other;

 Rational operator -() {
   return new Rational._normalized(-numerator, denominator);
 }

 Rational operator +(other) {
   if (other is Rational) {
     return new Rational(numerator * other.denominator
                             + other.numerator * denominator,
                         denominator * other.denominator);
   } else if (other is int) {
     return new Rational(numerator + other * denominator, denominator);
   } else {
     throw new ArgumentError("expecting Rational or int");
   }
 }

 Rational operator -(other) {
   return this + (-other);
 }

 Rational operator *(other) {
   if (other is Rational) {
     return new Rational(numerator * other.numerator,
                         denominator * other.denominator);
   } else if (other is int) {
     return new Rational(numerator * other, denominator);
   } else {
     throw new ArgumentError("expecting Rational or int");
   }
 }

 Rational operator /(other) {
   if (other is Rational) {
     return new Rational(numerator * other.denominator,
                         denominator * other.numerator);
   } else if (other is int) {
     return new Rational(numerator, denominator * other);
   } else {
     throw new ArgumentError("expecting Rational or int");
   }
 }

 double toDouble() => numerator / denominator;
}

final int hashCode #

Get a hash code for this object.

All objects have hash codes. Hash codes are guaranteed to be the same for objects that are equal when compared using the equality operator ==. Other than that there are no guarantees about the hash codes. They will not be consistent between runs and there are no distribution guarantees.

If a subclass overrides hashCode it should override the equality operator as well to maintain consistency.

bool operator ==(other) #

The equality operator.

The default behavior for all Objects is to return true if and only if this and other are the same object.

If a subclass overrides the equality operator it should override the hashCode method as well to maintain consistency.
